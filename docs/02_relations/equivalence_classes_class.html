Result name: equivalence_classes_class
Description: Prove that every element of A has an equivalence class

Source:

prove equivalence_classes_class: *A*R*C(is_equivalence_relation(R, A) & is_equivalence_classes(R, A, C) -> *X(X in A -> ^D(D in C & X in D))){ given |A, R, C|; implies equiv_relation, equiv_classes; extract expand(equiv_relation): reflexive, symmetric, transitive; given |X|; implies X_in_A; D_def = axiom_specify[<S: are_related(X, S, R)>](A)|D|; choose D; X_in_D = D_def(X)(X_in_A, expand(reflexive)(X)(X_in_A)); prove equiv_class_def: ^X(X in A & *Y(Y in D <-> Y in A & are_related(X, Y, R))){ choose X; prove right_def: *Y(Y in D <-> Y in A & are_related(X, Y, R)){ given |Y|; prove lemma0: Y in D -> Y in A & are_related(X, Y, R){ implies Y_in_D; extract D_def(Y)(Y_in_D): Y_in_A, related; return Y_in_A, related; } prove lemma1: Y in A & are_related(X, Y, R) -> Y in D{ implies premises; return D_def(Y)(premises); } return iff(lemma0, lemma1); } return X_in_A, right_def; } equiv_class = #is_equivalence_class(R, A, D)(equiv_class_def); D_in_C = expand(equiv_classes)(D)(equiv_class); return D_in_C, X_in_D; }
