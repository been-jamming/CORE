tuple_unique
Description:
prove tuple_unique: *A*B*C*D(is_tuple(A, B, C) & is_tuple(A, B, D) -> equal_sets(C, D)){ given |A, B, C, D|; implies both_tuples; extract both_tuples: tuple_C, tuple_D; tuple_C_def = expand(tuple_C); tuple_D_def = expand(tuple_D); extract tuple_C_def|E_C, F_C|: pair_E_C, pair_F_C, pair_C; extract tuple_D_def|E_D, F_D|: pair_E_D, pair_F_D, pair_D; E_C_equal_E_D = pair_unique(A, A, E_C, E_D)(pair_E_C, pair_E_D); F_C_equal_F_D = pair_unique(A, B, F_C, F_D)(pair_F_C, pair_F_D); pair_C_next = equality_consistency[<X: is_pair(X, F_C, C)>](E_C, E_D)(E_C_equal_E_D)(pair_C); pair_C_final = equality_consistency[<X: is_pair(E_D, X, C)>](F_C, F_D)(F_C_equal_F_D)(pair_C_next); return pair_unique(E_D, F_D, C, D)(pair_C_final, pair_D); }
